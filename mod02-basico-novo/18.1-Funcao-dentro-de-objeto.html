#18: Função dentro de objeto

        
        let pessoa = {
            nome:'Bonieky',
            sobrenome:'Lacerda',
            idade: 90,
            nomeCompleto: function(){
                // return this;
                // return `${pessoa.nome}  ${pessoa.sobrenome}, ${pessoa.idade}`;
                return `${this.nome}  ${this.sobrenome}, ${this.idade}`;
            },
            infoAdicional: () =>{
                return "Essa função anônima não tem acesso ao THIS"
            }
        }

        // console.log(`${pessoa.nome}  ${pessoa.sobrenome}, ${pessoa.idade}`);
        console.log(`${pessoa.nomeCompleto()}`);
        console.log(`${pessoa.infoAdicional()}`);
