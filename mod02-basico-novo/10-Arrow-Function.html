#10: Arrow Function

        // 1
        function somar_v1(x,y){
            let resultado = x+y;
            return resultado;
        }

        // 2
        const somar_v2 = (x,y) => x+y;

        console.log(somar_v1(2,3));
        console.log(somar_v2(2,3));

        // 3
        function sobrenome_1(sob) {
            return 'Boniek ' + sob;
        }

        // 4
        const sobrenome_2 = sob => 'Boniek ' + sob;

        console.log( sobrenome_1('Lacerda') );
        console.log( sobrenome_2('Lacerda') );
