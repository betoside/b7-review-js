        let esportes = {
            nome: 'surf',
            equipamento: 'prancha',
            icone: 'Medina',
            [Symbol.iterator]: function* (){
                yield this.nome;
                yield this.equipamento;
                yield this.icone;
            }
        }
        for(chave in esportes){
            console.log('chave:',chave);
        }
        for(valor of esportes){
            console.log('valor:',valor);
        }
