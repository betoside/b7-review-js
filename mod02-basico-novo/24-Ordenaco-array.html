        let frutas = ['Maçã', 'Uva', 'Laranja', 'Banana'];
        
        frutas.sort(); // ordem alfabética
        console.log(frutas);

        frutas.reverse(); // inverter a ordem do itens
        console.log(frutas);

        for(chave in frutas){
            console.log(`[${chave}]`,frutas[chave]);
        }

        let carros1 = [
            { marca: 'chevrolet', ano: 1984 },
            { marca: 'Fiat', ano: 1986 },
            { marca: 'Ford', ano: 1982 }
        ];
        console.log('carros1 original',carros1);

        // ordenar a lista pelo ano do carro
        carros1.sort();
        console.log('carros1 sort',carros1);

        let carros2 = [
            { marca: 'chevrolet', ano: 1984 },
            { marca: 'Fiat', ano: 1986 },
            { marca: 'Ford', ano: 1982 }
        ];
        // forma 1
        carros2.sort((a, b) => {
            if(a.ano > b.ano) {
                return 1;
            } else if(a.ano < b.ano) {
                return -1;
            } else {
                return 0;
            }
        });
        console.log('carros2 sort(a, b)',carros2);

        carros3 = [
            { marca: 'chevrolet', ano: 1984 },
            { marca: 'Fiat', ano: 1986 },
            { marca: 'Ford', ano: 1982 }
        ];

        // forma 2
        carros3.sort((a, b) => {
            // return b.ano - a.ano;
            return a.ano - b.ano;
        });

        console.log(carros3);


        // resumindo mais ainda a sintaxe
        carros4 = [
            { marca: 'chevrolet', ano: 1984 },
            { marca: 'Fiat', ano: 1986 },
            { marca: 'Ford', ano: 1982 }
        ];        

        carros4.sort((a, b) => b.ano - a.ano);
