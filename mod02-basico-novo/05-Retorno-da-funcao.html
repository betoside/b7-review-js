O propósito de uma função:

// ENTRADA -> PROCESSAMENTO -> SAÍDA 

        function nomeCompleto(nome, sobrenome) {
            return `${nome} ${sobrenome}`;
            console.log('Testando 123 se continua execução da função');
        }

        let completo = nomeCompleto('Jackson', 'Souza');
