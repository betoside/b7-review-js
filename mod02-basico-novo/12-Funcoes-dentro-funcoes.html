#12: Funções dentro de funções

* é legal usar arrow function para criar funções dentro de funções. 


        function addSquares(a,b) {

            function square(x) {
                return x*x;
            }

            let sqrA = square(a);
            let sqrB = square(b);
            return sqrA + sqrB;
        }


        function addSquares2(a,b) {

            const square = (x) => {
                return x*x;
            }

            let sqrA = square(a);
            let sqrB = square(b);
            return sqrA + sqrB;
        }

        console.log(addSquares(2,3));
        console.log(addSquares2(2,3));
