#25: Iteração de array 1

        
        let frutas = ['Maçã', 'Uva', 'Laranja', 'Banana'];
        console.log('frutas',frutas);

        let caracter_4 = frutas.filter((valor)=>{
            return valor.length > 4;
        });
        console.log('caracter_4', caracter_4);

        let maior_31 = frutas.every((valor)=>{
            return valor.length > 3;
        });
        console.log('nenhum item é > 3',maior_31);

        let maior_32 = frutas.some((valor)=>{
            return valor.length > 3;
        });
        console.log('algum item é > 3',maior_32);

        // descobrir se tem "Uva" dentro do array
        if(frutas.includes('Uva')){
            console.log('Tem uva ...');
        } else {
            console.log('Não tem uva ...');
        }
