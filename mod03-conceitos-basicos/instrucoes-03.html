    #8: Switch

        



        let _8dia = new Date();
        let _8diaNome = "";
        _8dia = _8dia.getDay();

        switch (_8dia) {
            case 1:
                _8diaNome = 'Segunda-feira';
                break;
        
            case 2:
            _8diaNome = 'Terça-feira';
            break;
    
            case 3:
                _8diaNome = 'Quarta-feira';
                break;
        
            case 4:
            _8diaNome = 'Quinta-feira';
            break;
    
            case 5:
                _8diaNome = 'Sexta-feira';
                break;
        
            case 6:
            _8diaNome = 'Sábado';
            break;
    
            case 0:
                _8diaNome = 'Domingo';
                break;
        }

        document.querySelector('._8dia').innerHTML = `Hoje é ${_8diaNome}`;
