#8: Manipulando atributos


        function senha(el) {
            password = document.querySelector('#senha');

            if( password.getAttribute('type') === 'password'){
                password.setAttribute('type', 'text');
                el.innerHTML = 'Esconder senha';
            } else {
                password.setAttribute('type', 'password');
                el.innerHTML = 'Mostrar senha';
            }
        }

        function clicou() {
            input = document.querySelector('input');

            console.log(input.getAttribute('type'));
            console.log(input.getAttribute('placeholder'));

            if (input.hasAttribute('placeholder')) {
                console.log('Tem placeholder');

                input.setAttribute('placeholder', 'placeholder alterado')
            } else {
                console.log('Não Tem placeholder');
            }
        }
