            function showtime(){
                let d = new Date();
                let h = d.getHours();
                let m = d.getMinutes();
                let s = d.getSeconds();

                if(h<10){
                    h = 0+h.toString();
                }
                if(m<10){
                    m = 0+m.toString();
                }
                if(s<10){
                    s = 0+s.toString();
                }

                let txt = h+':'+m+':'+s;

                document.querySelector('.demo13').innerHTML = txt;

                console.log(txt);
            }
