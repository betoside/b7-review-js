        let pessoa = {
            nome:'Boniek', 
            sobrenome:'Lacerda',
            idade:90,
            social:{
                facebook:'fb@b7web',
                instagram:'insta@boniek'
            },
            nomeCompleto: function(){
                return `${this.nome} ${this.sobrenome}`;
            }
        };

        console.log('nome:',pessoa.nome);
        console.log('rede social:', pessoa.social);
