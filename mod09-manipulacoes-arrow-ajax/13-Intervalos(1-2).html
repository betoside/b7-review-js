        function showtime(){
            let d = new Date();
            let h = d.getHours();
            let m = d.getMinutes();
            let s = d.getSeconds();

            let txt = h+':'+m+':'+s;

            demo.innerHTML = txt;
        }
        // showtime();

        let bt_comecar = document.querySelector('.btcomecar');
        let bt_parar = document.querySelector('.btparar');
        let demo = document.querySelector('.demo');


        bt_comecar.addEventListener('click', comecar);
        bt_parar.addEventListener('click', parar);

        let temporizador;

        function comecar(){
            temporizador = setInterval(showtime, 1000);
        }

        function parar(){
            clearInterval(temporizador);
            demo.innerHTML = '';
        }
