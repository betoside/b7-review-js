        function exe2_add(nomes, ...novosNomes){
            let novoConjunto = [
                ...nomes, // operador spread
                ...novosNomes // operador rest
            ];
            return novoConjunto;
        }

        let nomes2 = ['carlos','anselmo'];
        let outros2 = exe2_add(nomes2, 'artur','liliane','célia','joao olímpio');
