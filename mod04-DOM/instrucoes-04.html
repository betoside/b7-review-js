        function moverTela(horizontal, vertical){
            switch(horizontal){
                case "":
                case "0":
                case "inicio":
                    horizontal = 0;
                    break;
                case 'meio':
                    horizontal = (document.body.scrollWidth / 2);
                    break;
                case 'fim':
                    horizontal = document.body.scrollWidth;
                    break;
                default:
                    horizontal = calculaPosicaoDigitada('horizontal', horizontal);
                    break;
            }

            switch(vertical){
                case "":
                case "0":
                case "inicio":
                    vertical = 0;
                    break;
                case 'meio':
                    vertical = (document.body.scrollHeight / 2);
                    break;
                case 'fim':
                    vertical = document.body.scrollHeight;
                    break;
                default:
                    vertical = calculaPosicaoDigitada('vertical', vertical);
                    break;
            }

            // console.log('vertical', vertical);
            // console.log('horizontal', horizontal);
            // return;

            window.scrollTo({
                top: vertical,
                left: horizontal,
                behavior: "smooth"
            });
        }
        function calculaPosicaoDigitada(eixo_xy, posicao){
            
            if(eixo_xy == 'vertical'){
                referenciaDoCalculo = document.body.scrollHeight;
            } else {
                referenciaDoCalculo = document.body.scrollWidth;
            }

            if(posicao.indexOf('px') > 0){
                posicao = posicao.substring(0, posicao.indexOf('px'));

            } else {
                posicao = posicao.substring(0, posicao.indexOf('%'));
                posicao = (posicao / 100) * referenciaDoCalculo;
            }
            return posicao;
        }

            function posicionarTela(){
                let posicaoHorizontal = document.querySelectorAll('input[name=horiz]');
                let posicaoVertical = document.querySelectorAll('input[name=vert]');
                let horizFinal;
                let vertFinal;

                for (i in posicaoHorizontal) {

                    if( posicaoHorizontal[i].checked === true ){
                        let posicaoH = posicaoHorizontal[i].value;
                        let horiz_pos_espec = document.querySelector('input[name=horiz_pos_espec]');

                        if(  (posicaoH == 'px' || posicaoH == '%') && horiz_pos_espec.value.length > 0 ){
                            horizFinal = horiz_pos_espec.value.concat(posicaoH);

                        } else {
                            horizFinal = posicaoH;

                        }
                    }
                }


                for (i in posicaoVertical) {

                    if( posicaoVertical[i].checked === true ){
                        let posicaoV = posicaoVertical[i].value;
                        let vert_pos_espec = document.querySelector('input[name=vert_pos_espec]');

                        if(  (posicaoV == 'px' || posicaoV == '%') && vert_pos_espec.value.length > 0 ){
                            vertFinal = vert_pos_espec.value.concat(posicaoV);

                        } else {
                            vertFinal = posicaoV;

                        }
                    }
                }

                console.log('horizFinal:', horizFinal );
                console.log('vertFinal:', vertFinal );

                moverTela(horizFinal, vertFinal);

            }
