        class Person {

            static hands = 2;
            age = 0;
    
            constructor(name){
                this.name = name;
            }
    
            sayHi(){
                console.log(`Oi, eu sou ${this.name} e tenho ${Person.hands} mãos`);
            }
    
            static sayOi(){
                console.log(`Oi ${this.name}`);
            }
            
        }
    
        let p1 = new Person('Boniek');
        p1.sayHi();
        Person.sayOi();
